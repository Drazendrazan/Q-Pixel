        TriggerEvent('animations:client:EmoteCommandStart', {"bubbletea"})

        QBCore.Functions.Progressbar("drink_something", "Popping some Bubble Tea..", 5500, false, true, {
            disableMovement = false,
            disableCarMovement = false,
            disableMouse = false,
            disableCombat = true,
